<yt-player>

Reusable YouTube video displayer web component

Minimal

<yt-player playlist></yt-player>

Full featured

<yt-player playlist history></yt-player>

Multiple instances

<!-- Each has its own state and player -->

JavaScript API

// Get a reference
const player = document.querySelector('yt-player');

// Play a video
player.loadVideo('dQw4w9WgXcQ');

// Add to playlist
player.addToPlaylist('jNQXAC9IVRw');

// Read state
console.log(player.videoId);
console.log(player.playlistItems);

// Listen for events
player.addEventListener('play', e => console.log('Playing:', e.detail.id));
